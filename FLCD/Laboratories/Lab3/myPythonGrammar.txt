N = program statement_list statement declaration_stmt assignment_stmt io_stmt if_stmt while_stmt for_stmt relational_op condition expression expression_op term term_op factor identifier digit letter type vector_decl
E = null = < <= == != => > + - * / % ( ) { } } [ ] read print for while if else vector int char string constant
S = program
P =
program -> null | statement_list
statement_list -> statement | statement_list statement
statement -> declaration_stmt | assignment_stmt | io_stmt | if_stmt | while_stmt | for_stmt
declaration_stmt -> type identifier | type identifier = expression
assignment_stmt -> identifier = expression
io_stmt -> read ( identifier ) | print ( identifier )
if_stmt -> if condition { statement_list } | if condition { statement_list } else { statement_list }
while_stmt -> while condition { statement_list }
for_stmt -> for statement , condition , statement { statement_list }
condition -> expression relational_op expression
expression -> term | expression expression_op term
expression_op -> + | -
term -> factor | term term_op factor
term_op -> * | / | %
factor -> ( expression ) | identifier | constant
identifier -> _ | identifier letter | identifier digit
digit -> 0 | ... | 9
relational_op -> = | < | <= | == | != | => | >
type -> int | char | string | vector
vector_decl -> vector ( type ) identifier
letter -> a | ... | z | A | ... | Z
